    ```bash {{ title: 'cURL' }}
    curl --request GET \
        --url https://api.carisurau.com/surau \
        -H 'APP_KEY: {APP_KEY}' \
        -H 'APP_SECRET: {APP_SECRET}'

    ````

    ```js
    const options = {method: 'GET', headers: {APP_KEY: '', APP_SECRET: ''}};

    fetch('https://api.carisurau.com/surau', options)
        .then(response => response.json())
        .then(response => console.log(response))
        .catch(err => console.error(err));
    ```

    ```python
    import requests

    url = "https://api.carisurau.com/surau"

    payload = ""
    headers = {
        "APP_KEY": "",
        "APP_SECRET": ""
    }

    response = requests.request("GET", url, data=payload, headers=headers)

    print(response.text)
    ```

    ```php
    <?php

        $client = new http\Client;
        $request = new http\Client\Request;

        $request->setRequestUrl('https://api.carisurau.com/surau');
        $request->setRequestMethod('GET');
        $request->setHeaders([
        'APP_KEY' => '',
        'APP_SECRET' => ''
        ]);

        $client->enqueue($request)->send();
        $response = $client->getResponse();

        echo $response->getBody();
    ```
